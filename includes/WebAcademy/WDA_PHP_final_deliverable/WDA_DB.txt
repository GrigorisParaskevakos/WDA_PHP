CREATE DATABASE webAcademyDB;
USE webAcademyDB;
CREATE TABLE Contact (
User_id INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
Name VARCHAR(40) NOT NULL,
Email VARCHAR(60) NOT NULL,
Comment TEXT (140) NOT NULL,
Comment_date DATETIME NOT NULL,
PRIMARY KEY (User_id),
INDEX (Name),
INDEX (Comment_date)
) ENGINE=InnoDB;
